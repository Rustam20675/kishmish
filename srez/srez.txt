package Name

import "fmt"

func Kirishki() {
	/*Объяснение: Срез — это гибкий "вид" на массив. Это динамическая оболочка вокруг массива, которая может расти и уменьшаться. Срез состоит из трех частей:

	Указатель на лежащий в основе массив.
	Длина (len): сколько элементов срез содержит сейчас.
	Емкость (cap): сколько элементов массив может содержать максимально без перевыделения памяти.
	Особенности:

	Динамический: Размер можно менять с помощью append.
	Ссылочный тип: При присваивании передается не копия данных, а ссылка на один и тот же underlying array.*/
	// 1. Создание среза через литерал (Go сам создает массив в памяти)
	colors := [4]string{"Red", "Green", "Blue"}
	fmt.Println("Срез colors:", colors, "Длина:", len(colors), "Емкость:", cap(colors))

	// 2. Создание пустого среза с помощью make(тип, длина, емкость)
	// Емкость (cap) можно опустить, тогда она будет равна длине.
	nums := make([]int, 2, 5) // Длина=2, Емкость=5. [0, 0]
	fmt.Println("Срез nums до:", nums, "Длина:", len(nums), "Емкость:", cap(nums))

	// 3. Добавление элементов в срез (функция append)`
	nums = append(nums, 3)    // [0, 0, 3]`
	nums = append(nums, 4, 5) // Можно добавлять несколько элементов [0, 0, 3, 4, 5]`
	fmt.Println("Срез nums после append:", nums, "Длина:", len(nums), "Емкость:", cap(nums))

	// 4. Если добавить больше, чем вмещает емкость, Go автоматически создаст новый массив большего размера
	nums = append(nums, 6)
	fmt.Println("Срез nums после добавления 6-го:", nums, "Длина:", len(nums), "Емкость:", cap(nums)) // Емкость увеличилась (обычно в 2 раза)

	// 5. Создание среза из другого среза (или массива) [начальный_индекс:конечный_индекс]`
	// Важно: они будут указывать на один и тот же массив!`
	slice1 := []int{1, 2, 3, 4, 5}
	slice2 := slice1[1:4] // [2, 3, 4] (с 1-го индекса ДО 4-го, не включая его)
	fmt.Println("slice2:", slice2)

	slice2[0] = 999                                       // Меняем элемент в slice2...
	fmt.Println("slice1 после изменения slice2:", slice1) // ...и это меняет slice1! [1, 999, 3, 4, 5]

	// 6. Копирование срезов (чтобы избежать проблемы выше)
	source := []int{10, 20, 30}
	destination := make([]int, len(source))           // Создаем срез такой же длины
	copy(destination, source)                         // Копируем элементы из source в destination
	destination[0] = 999                              // Меняем копию...
	fmt.Println("Source остался неизменным:", source) // [10, 20, 30]
	fmt.Println("Destination изменен:", destination)  // [999, 20, 30]

	/*Объяснение: Mapa (или хеш-таблица) — это неупорядоченная коллекция пар ключ-значение. Ключи должны быть уникальными. Основная идея заключ в том что у нас есть ключь мы по нему считаем хэш складываем в бакеты .
	Когда бакеты в среднем заполнены больше чем на 6.5 элементов трегириться рост мапы и все элементы перемещаються в новые бакеты
	в бакеты могут находиться до 8ми элементов

	Особенности:

	Быстрый поиск, добавление и удаление элементов (в среднем за константное время O(1)).
	Порядок элементов не гарантируется.*/
	// 1. Объявление и инициализация карты
	// map[тип_ключа]тип_значения
	userAges := map[string]int{
		"Alice": 25,
		"Bob":   30, // Запятая в конце обязательна
	}
	fmt.Println("Карта целиком:", userAges)

	// 2. Добавление или изменение элемента
	userAges["Charlie"] = 28 // Добавили нового пользователя
	userAges["Alice"] = 26   // Изменили возраст Alice
	fmt.Println("После изменений:", userAges)

	// 3. Получение значения по ключу
	age := userAges["Bob"]
	fmt.Println("Возраст Bob:", age)

	// 4. Проверка существования ключа
	// При обращении по ключу возвращается 2 значения: (значение, bool)
	age, exists := userAges["David"]
	if exists {
		fmt.Println("Возраст David:", age)
	} else {
		fmt.Println("Ключ 'David' не найден в карте.") // Выполнится это
	}

	// 5. Удаление элемента по ключу (функция delete)
	delete(userAges, "Bob")
	fmt.Println("После удаления Bob:", userAges)

	// 6. Создание пустой карты с помощью make
	scores := make(map[string]int)
	scores["Math"] = 95
	scores["Kirech loh"] = 24
	fmt.Println("Оценки:", scores)
	delete(scores, "Kirech loh")
	fmt.Println("Оценки:", scores)
}
